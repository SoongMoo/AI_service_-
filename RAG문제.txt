from youtube_search import YoutubeSearch
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_community.document_loaders import YoutubeLoader
from langchain_ollama import ChatOllama

videos = YoutubeSearch("미국 대선", max_results=5).to_dict()
for v in videos:
    # url_suffix를 이용하여 video_url을 만듭니다.
    v['video_url'] = 'http://youtube.com' + v['url_suffix']

    # YoutubeLoader를 이용하여 비디오를 로드합니다.
    loader = YoutubeLoader.from_youtube_url(  # "자막(Transcript) 데이터"를 텍스트
        v['video_url'],
        language=['ko', 'en']
    )
    v['content'] = loader.load()


model = ChatOllama(
    model="gemma3:4b",
    temperature=0.7
)

prompt = ChatPromptTemplate.from_messages(
    [("system", "다음 영상에 대한 요약을 한국어로 만들어줘 :\n\n{context}")]
)

chain = prompt | model | StrOutputParser()

result = chain.invoke({"context": videos[0]['content']})
print(result)


위 코드는 Long Context Prompt방식으로 되어 있다. 
해당 코드를 RAG파이프라인 방식으로 변경하는데 있어 post-retriver의 10가지 중 제일 많이 사용되고 있는 FlashRankRerank, RedundantFilter, Document Compressor, Document Transformer4가지 방식도 같이 적용하여 
ipynb, py, txt로 되어 있는 파일을 highland0@nate.com 으로 보내주세오.
단, content를 만들 때 하나로 합쳐야 하며 꼭 chain을 사용할 수 있도록 한다. (감점 대상)

임베딩은 허깅체이스임베딩 모델을 사용하면됩니다.
